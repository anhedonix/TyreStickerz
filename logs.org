#+NAME: Graph-Logs-Formatter
#+BEGIN_SRC python :var input=Graph-Logs :cache yes :wrap SRC latex :exports results
  input = input.replace("*", "+")
  input = input.replace("\\ ", "\\textbackslash ")
  input = input.replace("\|", "\\textbackslash|")
  out = "\\begin{small}\n\\begin{Verbatim}[commandchars=\\\\\{\}]\n"
  data = []
  for line in input.splitlines():
    cline = str(line.strip()) + "\\\\"
    data.append(str(cline))
  out = out + "\n".join(data)+ "\n\\end{Verbatim}\n\\end{small}\n\\newpage"
  return out
#+END_SRC


#+NAME: Detailed-Logs-Formatter
#+BEGIN_SRC python :var input=Detailed-Logs :cache yes :wrap SRC latex :exports results
  input = input.replace("*", ".")
  out = "\\begin{itemize}\n"
  for line in input.splitlines()[3:]:
    cline = str(line.strip())
    if cline and cline[0] != "#":
      out = str(out) + "\n" + str(cline)
  out = out + "\n\\end{itemize}"
  return out
#+END_SRC
